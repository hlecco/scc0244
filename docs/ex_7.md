WHERE hemograma
AND de_analito = ANY(ARRAY['CHCM', 'Leucocitos', 'Linfocitos', 'Neutrofilos', 'Monocitos', 'Eosinofilos', 'Basofilos'])
GROUP BY de_exame, de_analito
